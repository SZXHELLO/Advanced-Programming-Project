# SDOC Editor 项目管理功能架构文档 v2.0

## 一、功能概述

为 SDOC Editor 增加基于角色的项目权限管理系统（RBAC），支持项目创建者邀请成员并分配不同权限级别，实现细粒度的协同编辑控制。

**v2.0 新增功能：**
- 邀请链接生成与分享机制
- 加入申请与审批流程
- 完整的项目活动日志系统

## 二、权限体系设计

### 2.1 角色定义

| 角色 | 代码标识 | 权限说明 |
|------|---------|---------|
| 管理员 | `admin` | 完全控制：编辑项目、邀请成员、分配权限、删除项目、审批申请 |
| 可编辑 | `editor` | 读写权限：查看和编辑项目内容，无法管理成员 |
| 只读 | `viewer` | 只读权限：仅查看项目内容，无法编辑或管理 |

### 2.2 权限矩阵

| 操作 | admin | editor | viewer |
|------|-------|--------|--------|
| 查看项目 | ✓ | ✓ | ✓ |
| 编辑标注 | ✓ | ✓ | ✗ |
| 修改页面 | ✓ | ✓ | ✗ |
| 导出 XML | ✓ | ✓ | ✓ |
| 邀请成员 | ✓ | ✗ | ✗ |
| 生成邀请链接 | ✓ | ✗ | ✗ |
| 审批加入申请 | ✓ | ✗ | ✗ |
| 修改权限 | ✓ | ✗ | ✗ |
| 删除项目 | ✓ | ✗ | ✗ |
| 查看活动日志 | ✓ | ✓ | ✓ |
| WebSocket 写操作 | ✓ | ✓ | ✗ |

## 三、数据库设计

### 3.1 新增数据表

#### Table: `project_members`
```sql
CREATE TABLE project_members (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  user_id INT NOT NULL,
  role ENUM('admin', 'editor', 'viewer') NOT NULL DEFAULT 'viewer',
  invited_by INT,
  invited_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  joined_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY unique_project_user (project_id, user_id),
  FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
  FOREIGN KEY (invited_by) REFERENCES users(id) ON DELETE SET NULL,
  INDEX idx_project (project_id),
  INDEX idx_user (user_id)
);
```

#### Table: `project_invitations`
```sql
CREATE TABLE project_invitations (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  inviter_id INT NOT NULL,
  invitee_username VARCHAR(100),  -- 用户名邀请时使用
  invitee_id INT,
  role ENUM('editor', 'viewer') NOT NULL DEFAULT 'viewer',
  status ENUM('pending', 'accepted', 'rejected', 'expired') DEFAULT 'pending',
  token VARCHAR(64) UNIQUE NOT NULL,
  invitation_type ENUM('username', 'link') DEFAULT 'username',  -- 新增：邀请类型
  max_uses INT DEFAULT 1,  -- 新增：最大使用次数（链接邀请）
  used_count INT DEFAULT 0,  -- 新增：已使用次数
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  expires_at DATETIME,
  responded_at DATETIME,
  FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE,
  FOREIGN KEY (inviter_id) REFERENCES users(id) ON DELETE CASCADE,
  FOREIGN KEY (invitee_id) REFERENCES users(id) ON DELETE CASCADE,
  INDEX idx_token (token),
  INDEX idx_invitee (invitee_id),
  INDEX idx_status (status),
  INDEX idx_type (invitation_type)
);
```

#### Table: `project_join_requests` (新增)
```sql
CREATE TABLE project_join_requests (
  id INT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  user_id INT NOT NULL,
  requested_role ENUM('editor', 'viewer') DEFAULT 'viewer',
  message TEXT,  -- 申请留言
  status ENUM('pending', 'approved', 'rejected') DEFAULT 'pending',
  reviewed_by INT,  -- 审批人
  review_message TEXT,  -- 审批意见
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  reviewed_at DATETIME,
  UNIQUE KEY unique_pending_request (project_id, user_id, status),
  FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
  FOREIGN KEY (reviewed_by) REFERENCES users(id) ON DELETE SET NULL,
  INDEX idx_project (project_id),
  INDEX idx_user (user_id),
  INDEX idx_status (status)
);
```

#### Table: `project_activity_logs` (新增)
```sql
CREATE TABLE project_activity_logs (
  id BIGINT AUTO_INCREMENT PRIMARY KEY,
  project_id INT NOT NULL,
  user_id INT,  -- 可为 NULL（系统操作）
  action_type VARCHAR(50) NOT NULL,  -- 操作类型
  action_detail JSON,  -- 操作详情（存储额外信息）
  target_user_id INT,  -- 操作目标用户（如移除成员时）
  ip_address VARCHAR(45),  -- IP 地址
  user_agent TEXT,  -- 浏览器信息
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL,
  FOREIGN KEY (target_user_id) REFERENCES users(id) ON DELETE SET NULL,
  INDEX idx_project (project_id),
  INDEX idx_user (user_id),
  INDEX idx_action (action_type),
  INDEX idx_created (created_at)
);
```

### 4.3 日志记录工具函数

#### 文件位置
`backend/utils/activityLogger.js`

```javascript
const db = require('../db');

/**
 * 记录项目活动日志
 * @param {Object} params
 * @param {number} params.projectId - 项目ID
 * @param {number} params.userId - 操作用户ID
 * @param {string} params.actionType - 操作类型
 * @param {Object} params.actionDetail - 操作详情
 * @param {number} params.targetUserId - 目标用户ID（可选）
 * @param {Object} params.req - Express request 对象（用于获取 IP 和 User-Agent）
 */
async function logActivity({
  projectId,
  userId,
  actionType,
  actionDetail = {},
  targetUserId = null,
  req = null
}) {
  const ipAddress = req ? (req.headers['x-forwarded-for'] || req.connection.remoteAddress) : null;
  const userAgent = req ? req.headers['user-agent'] : null;

  await db.query(
    `INSERT INTO project_activity_logs 
     (project_id, user_id, action_type, action_detail, target_user_id, ip_address, user_agent)
     VALUES (?, ?, ?, ?, ?, ?, ?)`,
    [projectId, userId, actionType, JSON.stringify(actionDetail), targetUserId, ipAddress, userAgent]
  );
}

module.exports = { logActivity };
```

**使用示例：**
```javascript
const { logActivity } = require('../utils/activityLogger');

// 在邀请成员时
await logActivity({
  projectId,
  userId: req.session.userId,
  actionType: 'member_invited',
  actionDetail: { inviteeUsername: username, role },
  req
});
```

### 4.4 权限验证中间件

#### 文件位置
`backend/middleware/projectAuth.js`

#### 核心函数

```javascript
// 检查用户是否为项目成员
async function checkProjectMember(req, res, next) {
  const { projectId } = req.params;
  const userId = req.session.userId;
  
  const [rows] = await db.query(
    'SELECT role FROM project_members WHERE project_id = ? AND user_id = ?',
    [projectId, userId]
  );
  
  if (rows.length === 0) {
    return res.status(403).json({ error: '无权访问此项目' });
  }
  
  req.userRole = rows[0].role;
  next();
}

// 检查是否为管理员
async function requireAdmin(req, res, next) {
  if (req.userRole !== 'admin') {
    return res.status(403).json({ error: '需要管理员权限' });
  }
  next();
}

// 检查是否有编辑权限
async function requireEditor(req, res, next) {
  if (!['admin', 'editor'].includes(req.userRole)) {
    return res.status(403).json({ error: '无编辑权限' });
  }
  next();
}
```

**修改文件：** `backend/server.js` 中的 `/ws/collab` 处理逻辑

**变更点：**

```javascript
const { logActivity } = require('./utils/activityLogger');

wss.on('connection', async (ws, req) => {
  const url = new URL(req.url, 'ws://localhost');
  const sessionId = url.searchParams.get('sessionId');
  const projectId = url.searchParams.get('projectId');
  
  // 1. 验证 session
  const session = await validateSession(sessionId);
  if (!session) {
    ws.close(4401, 'Unauthorized');
    return;
  }
  
  // 2. 验证项目成员权限
  const [rows] = await db.query(
    'SELECT role FROM project_members WHERE project_id = ? AND user_id = ?',
    [projectId, session.userId]
  );
  
  if (rows.length === 0) {
    ws.close(4403, 'Forbidden');
    return;
  }
  
  const userRole = rows[0].role;
  
  // 3. 监听消息，根据角色过滤写操作并记录日志
  ws.on('message', async (data) => {
    const msg = JSON.parse(data);
    
    // viewer 角色不能执行写操作
    if (userRole === 'viewer' && 
        ['annotation_add', 'annotation_update', 'annotation_delete', 'pages_replace'].includes(msg.type)) {
      ws.send(JSON.stringify({ 
        type: 'error', 
        message: '您是只读成员，无法编辑' 
      }));
      return;
    }
    
    // 记录操作日志
    if (msg.type === 'annotation_add') {
      await logActivity({
        projectId,
        userId: session.userId,
        actionType: 'annotation_added',
        actionDetail: {
          pageIndex: msg.pageIndex,
          text: msg.annotation?.text?.substring(0, 100) // 截取前100字符
        },
        req
      });
    } else if (msg.type === 'annotation_update') {
      await logActivity({
        projectId,
        userId: session.userId,
        actionType: 'annotation_updated',
        actionDetail: {
          annotationId: msg.annotationId,
          changes: msg.changes
        },
        req
      });
    } else if (msg.type === 'annotation_delete') {
      await logActivity({
        projectId,
        userId: session.userId,
        actionType: 'annotation_deleted',
        actionDetail: {
          annotationId: msg.annotationId
        },
        req
      });
    } else if (msg.type === 'pages_replace') {
      await logActivity({
        projectId,
        userId: session.userId,
        actionType: 'pages_replaced',
        actionDetail: {
          pageCount: msg.pages?.length
        },
        req
      });
    }
    
    // 正常广播逻辑...
  });
});
```

```javascript
// 审批申请
async function reviewRequest(requestId, action) {
  let grantedRole, message;
  
  if (action === 'approve') {
    grantedRole = document.getElementById(`granted-role-${requestId}`).value;
    message = prompt('审批意见（可选）：', '欢迎加入');
  } else {
    message = prompt('请说明拒绝理由：', '暂不符合加入条件');
    if (!message) return; // 用户取消
  }
  
  const resp = await fetch(`${API_BASE}/api/projects/${projectId}/join-requests/${requestId}/review`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${getSessionId()}`
    },
    body: JSON.stringify({ action, grantedRole, message })
  });
  
  if (resp.ok) {
    alert(action === 'approve' ? '已批准加入' : '已拒绝申请');
    loadJoinRequests();
    loadMembers(projectId); // 刷新成员列表
  } else {
    const err = await resp.json();
    alert(err.error);
  }
}

function getStatusText(status) {
  const map = {
    pending: '待审批',
    approved: '已批准',
    rejected: '已拒绝'
  };
  return map[status] || status;
}

      const newBadge = document.createElement('span');
      newBadge.className = 'badge';
      newBadge.textContent = count;
      document.getElementById('join-requests-tab').appendChild(newBadge);
    }
  } else if (badge) {
    badge.remove();
  }
}
```

```javascript
  if (!token) {
    showError('无效的邀请链接');
    return;
  }
  
  // 检查登录状态
  const sessionId = getSessionId();
  if (!sessionId) {
    document.getElementById('login-prompt').style.display = 'block';
    document.getElementById('loading').style.display = 'none';
    // 保存 token 到 sessionStorage，登录后跳转回来
    sessionStorage.setItem('pendingInvitationToken', token);
    return;
  }
  
  // 验证邀请链接
  try {
    const resp = await fetch(`${API_BASE}/api/invitation-links/${token}/info`, {
      headers: { 'Authorization': `Bearer ${sessionId}` }
    });
    
    if (resp.ok) {
      const data = await resp.json();
      displayInvitationInfo(data);
    } else {
      const err = await resp.json();
      showError(err.error || '邀请链接无效或已过期');
    }
  } catch (error) {
    showError('网络错误，请稍后重试');
  }
}

function displayInvitationInfo(data) {

  document.getElementById('invitation-info').style.display = 'block';
```

```javascript
function formatTime(dateString) {
  const date = new Date(dateString);
  const now = new Date();
  const diffMs = now - date;
  const diffMins = Math.floor(diffMs / 60000);
  
  if (diffMins < 1) return '刚刚';
  if (diffMins < 60) return `${diffMins} 分钟前`;
  
  const diffHours = Math.floor(diffMins / 60);
  if (diffHours < 24) return `${diffHours} 小时前`;
  
  const diffDays = Math.floor(diffHours / 24);
  if (diffDays < 7) return `${diffDays} 天前`;
  
  return date.toLocaleString('zh-CN');
}
```
